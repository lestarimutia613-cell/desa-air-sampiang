   430→    e.preventDefault();

   433→      const res = await fetch('/api/service-applications', {
   434→        method: 'POST',
   435→        headers: { 'Content-Type': 'application/json' },
   436→        body: JSON.stringify({
   437→          serviceType: config.serviceId,
   438→          applicantName: formData.namaLengkap || formData.namaPemilik || formData.namaBayi || formData.namaAlmarhum || Object.values(formData)[0] || '',

   445→      if (res.ok) {
   446→        setApplicationId(data.id ? data.id.slice(0, 8).toUpperCase() : Date.now().toString().slice(-8));
   447→        setSubmitted(true);

   449→        // Fallback: still show success for UX even if DB fails
   450→        setApplicationId(Date.now().toString().slice(-8));
   451→        setSubmitted(true);
   452→      }
   453→    } catch {

   457→      setSubmitting(false);
